# WHAT THIS FILE DOES:
# 1. Loads data/lost_found_dataset_cleaned.csv
# 2. Creates TEXT embeddings using sentence-transformers
#    (model: all-MiniLM-L6-v2 — fast, 384-dimensional)
# 3. Creates IMAGE embeddings using CLIP
#    (model: openai/clip-vit-base-patch32 — 512-dimensional)
# 4. Stores everything in ChromaDB (vector database)
# 5. Also saves embeddings to embeddings/lost_found_embeddings.pkl
#
# TOOLS USED: sentence-transformers, transformers (CLIP), chromadb, pandas

import pandas as pd
import numpy as np
import pickle
import os
import torch
from PIL import Image
from tqdm import tqdm

# Sentence-Transformers — text embeddings
from sentence_transformers import SentenceTransformer

# Transformers — CLIP image embeddings
from transformers import CLIPProcessor, CLIPModel

# ChromaDB — vector database
import chromadb
from chromadb.config import Settings

# ─────────────────────────────────────────────────────────────
# PATHS & CONFIG
# ─────────────────────────────────────────────────────────────
CLEAN_CSV       = "data/lost_found_dataset_cleaned.csv"
EMBEDDINGS_DIR  = "embeddings"
EMBEDDINGS_PKL  = "embeddings/lost_found_embeddings.pkl"
CHROMA_DIR      = "chroma_db"       # ChromaDB stores files here

# Model names
TEXT_MODEL_NAME  = "all-MiniLM-L6-v2"          # from sentence-transformers
IMAGE_MODEL_NAME = "openai/clip-vit-base-patch32"  # CLIP from HuggingFace

os.makedirs(EMBEDDINGS_DIR, exist_ok=True)
os.makedirs(CHROMA_DIR,     exist_ok=True)


# ─────────────────────────────────────────────────────────────
# STEP 1: Load models
# Both models download automatically on first run (~500MB total)
# ─────────────────────────────────────────────────────────────
def load_models():
    print("\n" + "="*60)
    print("PHASE 3: EMBEDDINGS & VECTOR DATABASE")
    print("="*60)

    # ── Text model (Sentence-Transformers) ──────────────────────
    print("\n[1/2] Loading Sentence-Transformers model...")
    print(f"      Model: {TEXT_MODEL_NAME}")
    print("      (Downloads ~90MB on first run)")
    text_model = SentenceTransformer(TEXT_MODEL_NAME)
    print(f"      ✅ Text model loaded — output dim: "
          f"{text_model.get_sentence_embedding_dimension()}")

    # ── Image model (CLIP) ─────────────────────────────────────
    print("\n[2/2] Loading CLIP model...")
    print(f"      Model: {IMAGE_MODEL_NAME}")
    print("      (Downloads ~600MB on first run)")
    clip_model     = CLIPModel.from_pretrained(IMAGE_MODEL_NAME)
    clip_processor = CLIPProcessor.from_pretrained(IMAGE_MODEL_NAME)
    clip_model.eval()  # set to inference mode
    print("      ✅ CLIP model loaded — output dim: 512")

    return text_model, clip_model, clip_processor


# ─────────────────────────────────────────────────────────────
# STEP 2: Create text embeddings using Sentence-Transformers
# Each row's search_text → 384-dimensional float vector
# ─────────────────────────────────────────────────────────────
def create_text_embeddings(df: pd.DataFrame,
                           model: SentenceTransformer) -> np.ndarray:
    """
    Converts all search_text strings to dense embedding vectors.

    sentence-transformers encodes meaning, so:
    "wireless earbuds" and "AirPods" get similar vectors
    even though they share no words.

    Returns: numpy array of shape (N, 384)
    """
    print("\n" + "-"*40)
    print("Creating TEXT embeddings (Sentence-Transformers)...")

    texts = df["search_text"].tolist()

    # encode() handles batching internally
    # show_progress_bar shows a nice tqdm bar
    embeddings = model.encode(
        texts,
        batch_size=32,
        show_progress_bar=True,
        convert_to_numpy=True,
        normalize_embeddings=True   # L2 normalise for cosine similarity
    )

    print(f"✅ Text embeddings shape: {embeddings.shape}")
    # Should be (number_of_items, 384)
    return embeddings


# ─────────────────────────────────────────────────────────────
# STEP 3: Create image embeddings using CLIP
# Each image → 512-dimensional float vector
# ─────────────────────────────────────────────────────────────
def embed_single_image(image_path: str,
                       model: CLIPModel,
                       processor: CLIPProcessor) -> np.ndarray:
    """
    Converts a single image to a CLIP embedding vector.
    Returns a 512-dimensional numpy array.
    """
    try:
        # Load image using Pillow
        image = Image.open(image_path).convert("RGB")

        # Preprocess for CLIP
        inputs = processor(images=image, return_tensors="pt")

        # Get image embedding — no gradient needed
        with torch.no_grad():
            features = model.get_image_features(**inputs)

        # Convert to numpy and L2-normalise
        emb = features.cpu().numpy().flatten()
        norm = np.linalg.norm(emb)
        return emb / norm if norm > 0 else emb

    except Exception as e:
        # Return zeros if image fails
        return np.zeros(512, dtype=np.float32)


def create_image_embeddings(df: pd.DataFrame,
                            clip_model: CLIPModel,
                            clip_processor: CLIPProcessor) -> np.ndarray:
    """
    Creates CLIP embeddings for all product images.
    Returns: numpy array of shape (N, 512)
    """
    print("\n" + "-"*40)
    print("Creating IMAGE embeddings (CLIP)...")

    embeddings = []
    for path in tqdm(df["image_path"].tolist(), desc="CLIP embedding"):
        emb = embed_single_image(path, clip_model, clip_processor)
        embeddings.append(emb)

    result = np.array(embeddings, dtype=np.float32)
    print(f"✅ Image embeddings shape: {result.shape}")
    # Should be (number_of_items, 512)
    return result


# ─────────────────────────────────────────────────────────────
# STEP 4: Store in ChromaDB
# ChromaDB is a persistent vector database.
# It stores embeddings + metadata and handles search for us.
# Data is saved in chroma_db/ folder on disk.
# ─────────────────────────────────────────────────────────────
def store_in_chromadb(df: pd.DataFrame,
                      text_embeddings: np.ndarray,
                      image_embeddings: np.ndarray):
    """
    Creates two ChromaDB collections:
      1. "lost_found_text"  — for text-based search
      2. "lost_found_image" — for image-based search

    ChromaDB stores vectors persistently in chroma_db/ folder.
    """
    print("\n" + "-"*40)
    print("Storing in ChromaDB (persistent vector database)...")
    print(f"  Database folder: {CHROMA_DIR}/")

    # Create a persistent ChromaDB client
    # PersistentClient saves everything to disk automatically
    client = chromadb.PersistentClient(path=CHROMA_DIR)

    # Delete existing collections if re-running
    for name in ["lost_found_text", "lost_found_image"]:
        try:
            client.delete_collection(name)
        except Exception:
            pass

    # ── Collection 1: Text embeddings ───────────────────────────
    print("\n  Creating text collection...")
    text_collection = client.create_collection(
        name="lost_found_text",
        metadata={"hnsw:space": "cosine"}   # use cosine similarity
    )

    # ChromaDB expects: ids (list of str), embeddings (list of list),
    # metadatas (list of dict), documents (list of str)
    text_collection.add(
        ids        = df["item_id"].tolist(),
        embeddings = text_embeddings.tolist(),
        metadatas  = df[["name", "category", "color", "price",
                         "price_bucket", "image_path",
                         "description", "lost_description"]
                        ].to_dict("records"),
        documents  = df["search_text"].tolist()
    )
    print(f"  ✅ Text collection: {text_collection.count()} items")

    # ── Collection 2: Image embeddings ──────────────────────────
    print("\n  Creating image collection...")
    image_collection = client.create_collection(
        name="lost_found_image",
        metadata={"hnsw:space": "cosine"}
    )

    image_collection.add(
        ids        = df["item_id"].tolist(),
        embeddings = image_embeddings.tolist(),
        metadatas  = df[["name", "category", "color", "price",
                         "price_bucket", "image_path",
                         "description", "lost_description"]
                        ].to_dict("records"),
        documents  = df["name"].tolist()
    )
    print(f"  ✅ Image collection: {image_collection.count()} items")
    print(f"\n  ChromaDB data saved in: {CHROMA_DIR}/")

    return client, text_collection, image_collection


# ─────────────────────────────────────────────────────────────
# STEP 5: Save embeddings to pickle file
# This allows the search engine to load everything quickly
# without re-running the embedding process
# ─────────────────────────────────────────────────────────────
def save_embeddings_pickle(df: pd.DataFrame,
                           text_emb: np.ndarray,
                           image_emb: np.ndarray):
    """
    Saves everything needed by the search engine to a pickle file.
    File: embeddings/lost_found_embeddings.pkl
    """
    payload = {
        "text_embeddings":  text_emb,
        "image_embeddings": image_emb,
        "dataframe":        df,
        "text_model_name":  TEXT_MODEL_NAME,
        "image_model_name": IMAGE_MODEL_NAME,
    }
    with open(EMBEDDINGS_PKL, "wb") as f:
        pickle.dump(payload, f)

    size_mb = os.path.getsize(EMBEDDINGS_PKL) / (1024 * 1024)
    print(f"\n✅ Embeddings saved → {EMBEDDINGS_PKL} ({size_mb:.1f} MB)")


# ─────────────────────────────────────────────────────────────
# STEP 6: Quick smoke test
# ─────────────────────────────────────────────────────────────
def smoke_test(text_collection, text_model):
    """
    Runs a quick test query to verify the database works.
    """
    print("\n" + "-"*40)
    print("SMOKE TEST: searching for 'gold wireless headphones'")

    query_emb = text_model.encode(
        ["lost gold wireless headphones"],
        normalize_embeddings=True
    )

    results = text_collection.query(
        query_embeddings=query_emb.tolist(),
        n_results=3
    )

    print("Top 3 matches:")
    for i, (meta, dist) in enumerate(
        zip(results["metadatas"][0], results["distances"][0])
    ):
        score = 1 - dist   # ChromaDB returns distance, convert to similarity
        print(f"  {i+1}. {meta['name']:<35} score={score:.4f}")


# ─────────────────────────────────────────────────────────────
# MAIN
# ─────────────────────────────────────────────────────────────
if __name__ == "__main__":
    # Load dataset
    df = pd.read_csv(CLEAN_CSV)
    print(f"Loaded {len(df)} items")

    # Load models
    text_model, clip_model, clip_processor = load_models()

    # Create embeddings
    text_emb  = create_text_embeddings(df, text_model)
    image_emb = create_image_embeddings(df, clip_model, clip_processor)

    # Store in ChromaDB
    client, text_col, image_col = store_in_chromadb(
        df, text_emb, image_emb
    )

    # Save pickle
    save_embeddings_pickle(df, text_emb, image_emb)

    # Test it
    smoke_test(text_col, text_model)

    print("\n" + "="*60)
    print("PHASE 3 COMPLETE!")
    print(f"  Embeddings pickle : {EMBEDDINGS_PKL}")
    print(f"  ChromaDB folder   : {CHROMA_DIR}/")
    print("="*60)
